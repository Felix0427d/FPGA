library ieee;
use ieee.std_logic_1164.all;
use ieee.numeric_std.all;
use ieee.math_real.all;

entity led is
    port (
        clk   : in std_logic;
        led_r : out std_logic
    );
end entity led;

architecture rtl of led is
    signal counter : unsigned(20 downto 0) := (others => '0');
begin
    -- Led output is open-drain, so duplicate this structure for the other LEDs
    led_r <= '0' when counter(counter'high) = '1' else 'Z';

    process (clk)
    begin
        if rising_edge(clk) then
            counter <= counter + 1;
        end if;
    end process;

end architecture;
